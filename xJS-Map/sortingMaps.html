Examples of how to sort Map

Below we will generate the Map, which will use for examples

Source code:

let exampleMap = new Map()
exampleMap.set("Lemon", new Map())
exampleMap.set("Peach", new Map())
exampleMap.set("Apple", new Map())
exampleMap.set("Banana", new Map())
exampleMap.set("Peach").set("January", 1)
exampleMap.set("Peach").set("April", 2)
exampleMap.set("Peach").set("March", 3)
exampleMap.set("Peach").set("July", 9)
exampleMap.set("Peach").set("September", 4)
exampleMap.set("Peach").set("December", 11)
exampleMap.set("Lemon").set("Sunday", 1)
exampleMap.set("Lemon").set("Tuesday", 2)
exampleMap.set("Lemon").set("Friday", 3)
exampleMap.set("Banana").set("Year", 2018)
exampleMap.set("Apple").set("dogs", 11)
exampleMap.set("Apple").set("cats", 12)
exampleMap.set("Apple").set("mouses", 3)

Code output:

Map { 'Lemon' => Map { 'Sunday' => 1, 'Tuesday' => 2, 'Friday' => 3 }, 'Peach' => Map { 'January' => 1, 'April' => 2, 'March' => 3, 'July' => 9, 'September' => 4, 'December' => 11 }, 'Apple' => Map { 'dogs' => 11, 'cats' => 12, 'mouses' => 3 }, 'Banana' => Map { 'Year' => 2018 } }

Sorted keys in the map

Here we get only the keys sorted alphabetically

Source code:

let sortedMapKeys = Array.from(exampleMap.keys()).sort()

Code output:

[ 'Apple', 'Banana', 'Lemon', 'Peach' ]

Values in the map sorted by keys

We will get all values in the map sorted by their keys. Note that we dont't actually change the Map, we just get its keys sorted in an array and use the array to access them in the map, but the Map actually stays unchanged.

Source code:

let sortedMapKeys = Array.from(exampleMap.keys()).sort() for(const key of sortedMapKeys) { console.log(exampleMap.get(key)) }

Code output:

Map { 'dogs' => 11, 'cats' => 12, 'mouses' => 3 } Map { 'Year' => 2018 } Map { 'Sunday' => 1, 'Tuesday' => 2, 'Friday' => 3 } Map { 'January' => 1, 'April' => 2, 'March' => 3, 'July' => 9, 'September' => 4, 'December' => 11 }

Nested keys of Map sorted by main keys

Here we will sort keys of the object and then sort nested keys and access them

Source code:

let sortedMapKeys = Array.from(exampleMap.keys()).sort() for(const key of sortedMapKeys) { let sortedNestedKeys = Array.from(exampleMap.get(key).keys()).sort() console.log(sortedNestedKeys) }

Code output:

[ 'cats', 'dogs', 'mouses' ] [ 'Year' ] [ 'Friday', 'Sunday', 'Tuesday' ] [ 'April', 'December', 'January', 'July', 'March', 'September' ]

Nested keys of Map sorted by their values

Here we first create array of the Map keys and use those keys to get the keys for the nested Maps. We use nested values to sort the nested keys and for equal values we sort them by the nested alphabetical key

Source code:

let sortedMapKeys = Array.from(exampleMap.keys()).sort() for(const key of sortedMapKeys) { let sortedNestedKeys = Array.from(exampleMap.get(key).keys()).sort((a, b) => { let difference = exampleMap.get(key).get(a) - exampleMap.get(key).get(b) if (difference === 0) { return a.localeCompare(b) } return difference }) console.log(sortedNestedKeys) }

Code output:

[ 'mouses', 'dogs', 'cats' ] [ 'Year' ] [ 'Sunday', 'Tuesday', 'Friday' ] [ 'January', 'April', 'March', 'September', 'July', 'December' ]
